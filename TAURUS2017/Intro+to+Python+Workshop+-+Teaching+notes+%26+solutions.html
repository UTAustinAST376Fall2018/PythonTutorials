# Exercise 4 solutions

def gauss(mu, sig, x):
    const = (sig**2 * np.sqrt(2*np.pi))
    exp = np.exp(-(x-mu)**2/(2*sig**2))
    return const*exp

xarr = np.linspace(-5, 5, 1000)

gaussian = gauss(0, 1, xarr)

import matplotlib.pyplot as plt
plt.plot(xarr, gaussian)
plt.show()

from scipy.stats import norm

sig2 = np.random.normal(0, 2, 10000)
sig15 = np.random.normal(0, 1.5, 10000)

plt.hist(sig2, bins=50, hatch='|', alpha=0.5, facecolor='magenta', label='$\sigma=2$', normed=True)
plt.hist(sig15, bins=50, alpha=0.25, facecolor='cyan', label='$\sigma=1.5$', normed=True)

x=np.linspace(-6, 6, 10000)
mu2, sigma2 = norm.fit(sig2)
p2 = norm.pdf(x, mu2, sigma2)
plt.plot(x, p2, 'm', linewidth=2)
mu15, sigma15 = norm.fit(sig15)
p15 = norm.pdf(x, mu15, sigma15)
plt.plot(x, p15, 'c', linewidth=1.5)

plt.legend()
plt.show()
